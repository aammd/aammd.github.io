What this post is about

I’m passionate about ordered contrasts. I think that very often ecologists have experimental treatments that are in the form of three levels: less than normal, normal amount, more than normal amount

lets program this into Stan

lil_fac <- letters[1:3]
lil_fac <- ordered(lil_fac)
ord_contr <- cbind(1, contrasts(lil_fac))
dimnames(ord_contr) <- NULL
# dput(ord_contr)
ordinal_mean_prior <- cmdstan_model(here::here("posts/2023-08-29-ordinal-moments/ordinal_mean_prior.stan"))

ordinal_mean_prior
data {
  int n_each;
}
transformed data {
  int n_observations = n_each * 3;
  matrix[3, 3] contr = [
    [1, -0.707106781186548, 0.408248290463863],
    [1, -7.85046229341888e-17, -0.816496580927726],
    [1, 0.707106781186547, 0.408248290463863]
    ];
}
parameters {
  vector[3] betas;
  real<lower=0> sigma;
  vector[n_observations] obs;
}
transformed parameters {
  vector[3] m = contr * betas;
  vector[n_observations] mu;
  for (i in 1:n_each){
    mu[i] = m[1];
    mu[i + n_each] = m[2];
    mu[i + 2*n_each] = m[3];
  }
}
model{
  betas ~ std_normal();
  sigma ~ exponential(1);
  obs ~ normal(mu, sigma);
}
ordinal_mean_prior_predictive <- ordinal_mean_prior$sample(data = list(n_each = 10))

ordinal_grpid_prior
data {
  int n;
  array[n] int<lower=1, upper=3> group_id;
}
transformed data {
  matrix[3, 3] contr = [
    [1, -0.707106781186548, 0.408248290463863],
    [1, -7.85046229341888e-17, -0.816496580927726],
    [1, 0.707106781186547, 0.408248290463863]
    ];
}
parameters {
  vector[3] betas;
  real<lower=0> sigma;
  vector[n] obs;
}
transformed parameters {
  vector[3] m = contr * betas;
  vector[n] mu = m[group_id];
}
model{
  betas ~ std_normal();
  sigma ~ exponential(1);
  obs ~ normal(mu, sigma);
}

condition on (fake) observations

its the same model as before except we move obs up to the data block:

ordinal_contrasts <- cmdstan_model(here::here("posts/2023-08-29-ordinal-moments/ordinal_contrasts.stan"))

ordinal_contrasts
data {
  int n;
  array[n] int<lower=1, upper=3> group_id;
  vector[n] obs;
}
transformed data {
  matrix[3, 3] contr = [
    [1, -0.707106781186548,     0.408248290463863],
    [1, -7.85046229341888e-17, -0.816496580927726],
    [1,  0.707106781186547,     0.408248290463863]
    ];
}
parameters {
  vector[3] betas;
  real<lower=0> sigma;
}
transformed parameters {
  vector[3] m = contr * betas;
  vector[n] mu = m[group_id];
}
model{
  betas ~ std_normal();
  sigma ~ exponential(1);
  obs ~ normal(mu, sigma);
}
generated quantities {
  vector[3] yrep;
  for (k in 1:3){
    yrep[k] = normal_rng(m[k], sigma);
  }
}

concrete example

Let’s consider the growth rate of seedlings in proportion to the distance from a parent plant. We’d expect this to decline as you get farther away, and probably in a nonlinear way..

true_values <-  c(5, -3, 1.1)
gr_mu <- ord_contr %*% true_values

plot(gr_mu)

See how easy that was to make!? that’s why this works! ordered contrasts are easier to think about.

growth_data <- tibble(group_id = rep(1:3, each = 30), 
                      true_mean = gr_mu[group_id],
                      distance = c("1m", "5m", "20m")[group_id],
                      obs = rnorm(length(true_mean), mean = true_mean, sd = 1.1))

growth_data |> 
  ggplot(aes(x = distance, y = obs)) + geom_point()

fit it with the same model

ordinal_contrasts_post <- ordinal_contrasts$sample(data = list(n = nrow(growth_data),

ordinal_contr_mu_sd <- cmdstan_model(here::here("posts/2023-08-29-ordinal-moments/ordinal_contr_mu_sd.stan"))

ordinal_contr_mu_sd
data {
  int n;
  array[n] int<lower=1, upper=3> group_id;
  vector[n] obs;
}
transformed data {
  matrix[3, 3] contr = [
    [1, -0.707106781186548,     0.408248290463863],
    [1, -7.85046229341888e-17, -0.816496580927726],
    [1,  0.707106781186547,     0.408248290463863]
    ];
}
parameters {
  vector[3] mu_beta;
  vector[3] sigma_beta;
}
transformed parameters {
  vector[3] m = contr * mu_beta;
  vector[3] s = exp(contr * sigma_beta);
  vector[n] mu = m[group_id];
  vector[n] sigma = s[group_id];
}
model{
  mu_beta ~ std_normal();
  sigma_beta ~ std_normal();
  sigma ~ exponential(1);
  obs ~ normal(mu, sigma);
}
generated quantities {
  vector[3] yrep;
  for (k in 1:3){
    yrep[k] = normal_rng(m[k], s[k]);
  }
}
